\(                          #普通字符“(”
    (                     #分组构造，用来限定量词“*”修饰范围
        [^()]+              #非括弧的其它任意字符
    |                       #分支结构
        \(  (?<Open>)       #命名捕获组，遇到开括弧Open计数加1
    |                       #分支结构
        \)  (?<-Open>)      #狭义平衡组，遇到闭括弧Open计数减1
    )*                      #以上子串出现0次或任意多次
    (?(Open)(?!))           #判断是否还有'OPEN'，有则说明不配对，什么都不匹配
\)                          #普通闭括弧
